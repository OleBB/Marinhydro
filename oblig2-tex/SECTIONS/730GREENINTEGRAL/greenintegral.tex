Vi formulerer grenseverdiproblemet med Green-funksjonen, der
\begin{align}
	g(x,y; \bar{x}, \bar{y}, t) = Re\big( G(x,y; \bar{x}, \bar{y}, t) e^{\mathrm{i} \omega t} \big),\\
	G(x,y; \bar{x}, \bar{y}, t) =\log r + H(x,y; \bar{x}, \bar{y}, t), \\ 
	r = \sqrt{(x- \bar{x})^2 + (y- \bar{y})^2}
\end{align}
